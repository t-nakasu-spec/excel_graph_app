    function updateSelectionStyles(qIdx, isMulti) {
        const container = $(`q${qIdx}_container`);
        const labels = container.querySelectorAll('.choice');
        labels.forEach(l => {
            if(l.querySelector('input').checked) {

            } else {
                l.classList.remove('choice-selected');

        });

    function restoreAnswer(qIdx, q) {
      const a = answers[qIdx];
      if (!a) return;

      if (q.type === 'é¸æŠ') {
        const inputs = document.querySelectorAll(`input[name="q${qIdx}"]`);
        inputs.forEach(inp => {
          if (Array.isArray(a.response)) {
            if (a.response.includes(inp.value)) inp.checked = true;
          } else {
            if (a.response === inp.value) inp.checked = true;
          }
        });
        updateSelectionStyles(qIdx, q.multi);
      } else {
        const ta = $(`q${qIdx}_free`);
        if(ta) ta.value = a.response;
      }
    }

    function saveCurrentPage() {
  const start = pageIndex * PAGE_SIZE;
  for (let i = 0; i < PAGE_SIZE; i++) {
    const qIdx = start + i;
    if (qIdx >= questions.length) break;
    
    const q = questions[qIdx];
    let res = null;

    if (q.type === 'é¸æŠ') {
      const selected = Array.from(document.querySelectorAll(`input[name="q${qIdx}"]:checked`)).map(el => el.value);
      if (selected.length) {
        res = q.multi ? selected : selected[0];
      }
    } else {
      const ta = $(`q${qIdx}_free`);
      if (ta) {
        const val = ta.value.trim();
        if (val) res = val;
      }
    }

    // å›ç­”ãŒã‚ã‚‹å ´åˆã®ã¿ä¿å­˜ã€ãªã„å ´åˆã¯nullã®ã¾ã¾ï¼ˆä¸Šæ›¸ãã—ã¦æ¶ˆã•ãªã„ã‚ˆã†ã«æ³¨æ„ï¼‰
    if (res !== null) {

    }
  }
